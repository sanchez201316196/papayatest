function game1($arr, $N){
	//$arr = [5,2,3,-1,11];
	//$N = 13;
	
	$result = [];
	for($i = 0; $i < count($arr) - 1; $i++){
		$ini = $arr[$i];
		$resto = $N - $ini;
		for($j = $i+1; $j < count($arr); $j++){
			if($resto == $arr[$j]){
				array_push($result,$ini ,$arr[$j]);
				break 2;
			}
		}
	}
	
	return $result;
	
	//var_dump($result);
}
